$.mergeData(target,src)

描述：

合并src的缓存数据到target的缓存数据上。

参数：

target
必需。目标对象。
src
可选。被合并数据的对象。
例子 
require("data,ready", function($) {
    var obj1 = {};
    //写入数揣
    $.data(obj1, "name", "司徒正美");
    //读取数据
    $.log($.data(obj1, "name"), true); //司徒正美
    $._data(obj1, "inner", "私有数据");
    var obj2 = {};
    $.data(obj2, "sex", "man");
    //合并数据
    $.mergeData(obj1, obj2);
    $.log( $.dump($._data(obj1)), true); 
});
