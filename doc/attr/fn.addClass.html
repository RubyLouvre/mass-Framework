$.fn.addClass(className)

描述：

为每个匹配的元素添加指定的类名.

参数：

className
必需。String。如果要添加多个类名，请用空字符隔开，如'aaa bbb ccc'
例子
aaaaa
bbbbb
ccccc

                if (!el.className) {
                    el.className = item;

addClass: function(item) {
    if (typeof item == "string") {
        for (var i = 0, el; el = this[i++];) {
            if (el.nodeType === 1) {
                if (!el.className) {
                    el.className = item;
                } else {

                    a.sort();
                    for (var i = a.length - 1; i > 0; --i)
                    if (a[i] == a[i - 1]) a.splice(i, 1);
