            $.require("ready,more/droppable",function(){
                $('.drag').droppable({
                    range:".drop",
                    dragover: function(event, dd){
                        dd.dragger.text(dd.dropper.text())
                    },
                    dragleave: function(event, dd){
                        dd.dragger.text("")
                    },
                    drop: function(event, dd){
                        dd.dropper.toggleClass("dropped");
                    }
                })
            });
