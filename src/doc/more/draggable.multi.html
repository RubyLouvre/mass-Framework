通过点击选中一起要拖动的方块，再拖动它们。 返回

            $.require("ready,more/draggable",function(){
                $(".drag").draggable({
                    dragstart:function(){
                        $( this ).addClass("active");
                    },
                    dragend:function(){
                        $( this ).removeClass("active");
                    }
                })
            });
